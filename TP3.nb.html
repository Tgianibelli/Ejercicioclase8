vinosd <- vinos %>% 
  group_by(variedad) %>% 
  filter(n() > 3000) %>% 
  mutate(puntaje = as.character
         (cut(puntos, 
              breaks = c(80, 85, 90, 95, 100), 
              include.lowest = TRUE, 
              labels = c('80-85', '86-90', '91-95', '96-100')))) %>% 
  select(c('variedad', 'puntaje')) %>% 
  mutate(value = n()) %>% 
  gather_set_data(1:2)
